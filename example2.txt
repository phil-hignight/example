// AI Agent - Built 2025-11-17T18:42:39.414Z
(function() {
  window.module = {};

  // Module: main.js
  // Main entry point for the app
  window.module['main.js'] = {
    run: async function() {
      // Clear existing page content
      document.body.innerHTML = '';
  
      // Create container
      const container = document.createElement('div');
      container.id = 'ai-agent-container';
      container.style.cssText = 'width: 100%; max-width: 800px; margin: 50px auto; padding: 20px; font-family: monospace;';
  
      // Create title
      const title = document.createElement('h1');
      title.textContent = 'AI Agent';
      title.style.cssText = 'color: #333; margin-bottom: 20px;';
      container.appendChild(title);
  
      // Create output div
      const output = document.createElement('div');
      output.id = 'output';
      output.style.cssText = 'white-space: pre-wrap; font-size: 14px; line-height: 1.6; background: #f5f5f5; padding: 20px; border-radius: 4px;';
      container.appendChild(output);
  
      document.body.appendChild(container);
  
      output.textContent = 'Hello from the AI UI app!\n\n';
      output.textContent += 'Starting API test...\n';
  
      try {
        // Get auth token from localStorage (simulate production)
        const authToken = localStorage.getItem('token');
  
        if (!authToken) {
          output.textContent += '\nWARNING: No auth token found. Setting a test token...\n';
          localStorage.setItem('token', 'test-token-123');
        }
  
        const apiUrl = '/api/chat/completions';
  
        const requestBody = {
          stream: true,
          model: 'Anthropic Claude 4 Sonnet',
          messages: [
            {
              role: 'user',
              content: 'Say hello and tell me a short joke!'
            }
          ],
          params: {},
          features: {
            web_search: false
          },
          session_id: 'test-session-' + Math.random().toString(36).substr(2, 9),
          id: 'msg-' + Date.now() + '-assistant',
          background_tasks: {
            title_generation: false,
            tags_generation: false
          }
        };
  
        output.textContent += '\nSending request to: ' + apiUrl + '\n';
        output.textContent += 'Request: ' + JSON.stringify(requestBody.messages[0], null, 2) + '\n\n';
  
        const headers = {
          'Content-Type': 'application/json'
        };
  
        const token = localStorage.getItem('token');
        if (token) {
          headers['Authorization'] = `Bearer ${token}`;
        }
  
        const response = await fetch(apiUrl, {
          method: 'POST',
          headers: headers,
          body: JSON.stringify(requestBody)
        });
  
        if (!response.ok) {
          const errorText = await response.text();
          output.textContent += `\nERROR: API request failed (${response.status}): ${errorText}\n`;
          return;
        }
  
        output.textContent += 'Response received! Streaming...\n';
        output.textContent += '---\n';
  
        // Read streaming response
        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        let finalContent = '';
  
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;
  
          const chunk = decoder.decode(value);
          const lines = chunk.split('\n');
  
          for (const line of lines) {
            if (line.startsWith('data: ')) {
              const data = line.slice(6).trim();
              if (data === '[DONE]') {
                continue;
              }
  
              if (!data) continue;
  
              try {
                const parsed = JSON.parse(data);
                if (parsed.choices && parsed.choices[0] &&
                    parsed.choices[0].delta && parsed.choices[0].delta.content) {
                  const content = parsed.choices[0].delta.content;
                  finalContent += content;
                  output.textContent += content;
                }
              } catch (e) {
                console.warn('Error parsing SSE data:', e);
              }
            }
          }
        }
  
        output.textContent += '\n---\n';
        output.textContent += '\nAPI test completed successfully!';
        output.textContent += '\nTotal characters received: ' + finalContent.length;
  
      } catch (error) {
        output.textContent += '\n\nERROR: ' + error.message;
        console.error('Error:', error);
      }
    }
  };
  

  // Run main module
  if (window.module['main.js']) {
    window.module['main.js'].run();
  } else {
    console.error('Error: main.js module not found');
  }
})();
